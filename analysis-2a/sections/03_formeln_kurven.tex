\subsection{Tangenten}
\textbf{Tangentengleichung:} $y = m \, (x - x_0) + y_0$ \qquad $m$ = Steigung \textrightarrow\ Berechung gemäss Kapitel \ref{Übersicht -- Kurvenformeln}


\subsubsection{Berechnung Tangentensteigung bei impliziter Kurvenformel}

\textrightarrow\ Implizite Differentiation! \qquad \textbf{Ziel:} $f'(x) = y' = \dfrac{\diff y}{\diff x}$

\medskip

\begin{minipage}[t]{0.48\columnwidth}
	\example{Kreis}

	\renewcommand{\arraystretch}{2.2}
	\begin{tabular}{l}
		$x^2 + y^2 = 4 \quad \Big| \diff(\ldots)$ Differential 			\\
		$2x \, \diff x + 2y \, \diff y = 0 \quad \Big| \div \, \diff x$	\\
		$2x + 2y \, \dfrac{\diff y}{\diff x} = 0$ 							\\
		$\dfrac{\diff y}{\diff x} = - \dfrac{2x}{2y} = - \dfrac{x}{y}$
	\end{tabular}
	\renewcommand{\arraystretch}{1}
\end{minipage}		
\hfill
\begin{minipage}[t]{0.48\columnwidth}
	\example{Mit Produktregel}

	\renewcommand{\arraystretch}{2.2}
	\begin{tabular}{l}
		$y^3 = 2 x^2 + 5 xy \quad \Big| \diff(\ldots)$ Differential 													\\ 
		$3 y^2 \, \diff y = 4x \, \diff x + 5(1y + xy') \, \diff x \quad \Big| \div \, \diff x$ 						\\ 
		$3 y^2 \cbl{\dfrac{\diff y}{\diff x}} = 4x + 5(y + xy') \; \Leftrightarrow \; 3 y^2 \cbl{y'} = 4x + 5y + 5xy'$	\\
		$3 y^2 y' - 5 xy' = 4x + 5y \; \Leftrightarrow \; y'(3y^2 - 5x) = 4x + 5y$ 										\\
		$y' = f'(x) = \dfrac{4x + 5y}{3y^2 - 5x}$ 
	\end{tabular}
